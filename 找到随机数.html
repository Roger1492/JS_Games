随机数:***

输入你猜测的数:


            // 源代码
            window.onload = function () {
                let rndNum = document.getElementById("rndNum");
                let text = document.getElementsByTagName("input")[0];
                let result = document.getElementById("result");
                let index = 0;
                let a = Math.floor(Math.random()*100);
    
                text.addEventListener("keypress", function(e){
                    if(e.keyCode === 13){
                        if(Number(this.value) > Number(a)){
                            ++index;
                            result.innerHTML = "猜测的结果比随机数大，请输入小一点的数字";
                            document.getElementById("small").style.fontSize = "24px";
                            document.getElementById("small").style.color = "red";
                        } else if(Number(this.value) < Number(a)){
                            ++index;
                            result.innerHTML = "猜测的结果比随机数大，请输入大一点的数字";
                            document.getElementById("big").style.fontSize = "24px";
                            document.getElementById("big").style.color = "blue";
                        } else {
                            ++index;
                            rndNum.innerHTML = a;
                            result.innerHTML = "厉害，你只猜了 " + index + " 次就猜对了,答案就是: " + a;
                            result.style.color = "green";
                            result.style.fontSize = "30px";
                        }
                        text.value = null;
                    }
                }, false);
            };
